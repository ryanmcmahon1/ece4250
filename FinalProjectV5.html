# returns average value of the 5-neighborhood of a pixel at x,y. If there is
# an out of bounds error, the value isn't included. The sum and number of 
# elements counted is also returned
def w2_avg(img, x, y, xmax, ymax):
    
    # variables that store the number of pixels we are including in the
    # average, and the total sum. used to make sure we don't have out of
    # bounds accesses
    num = 0
    sum = 0
    
    sum += img[x][y]
    num += 1
    
    if (x+1 < xmax):
        sum += img[y][x+1]
        num += 1
    
    if (x-1 >= 0):
        sum += img[y][x-1]
        num += 1
    
    if (y+1 < ymax):
        sum += img[y+1][x]
        num += 1
    
    if (y-1 >= 0):
        sum += img[y-1][x]
        num += 1
        
    return sum/num, sum, num

def lex(img):
    
    # determine num_rows and num_cols, which should be 256x256 in this case
    rows = img.shape[0]
    cols = img.shape[1]
    
    # arrays that store average intensity values for neighborhoods
    # surrounding each pixel
    lex1 = np.zeros(rows*cols)
    lex2 = np.zeros(rows*cols)
    lex3 = np.zeros(rows*cols)
    lex4 = np.zeros(rows*cols)
    lex5 = np.zeros(rows*cols)
    lex6 = np.zeros(rows*cols)
    
    ct = 0
    for y in np.arange(rows):
        for x in np.arange(cols):
            
            # w1: just the pixel value itself
            lex1[ct] = img[y][x]
            
            # w2: pixel, and four surrounding pixels 
            w2 = w2_avg(img, x, y, cols, rows)
            lex2[ct] = w2[0]
            
            # w3: pixel, and eight surrounding pixels
            w3 = w3_avg(img, x, y, cols, rows, w2[1], w2[2])
            lex3[ct] = w3[0]
            
            # w4: pixel, and 12 surrounding pixels
            w4 = w4_avg(img, x, y, cols, rows, w3[1], w3[2])
            lex4[ct] = w4[0]
            
            # w5: pixel, and 20 surrounding pixels
            w5 = w5_avg(img, x, y, cols, rows, w4[1], w4[2])
            lex5[ct] = w5[0]
            
            # w6: pixel, and 24 surrounding pixels
            w6 = w6_avg(img, x, y, cols, rows, w5[1], w5[2])
            lex6[ct] = w6
            
            # increment count variable
            ct = ct + 1
            
    return lex1,lex2,lex3,lex4,lex5,lex6

# given a linear index, returns (x,y) (only works for 256x256 images)
def x_y(i):
    y = np.floor(i/256)
    x = i - 256*y
    return int(x),int(y)

# takes in array that gives sorted pixels, returns histogram
def exact_hist(img, arr):
    
    # local variables to help with assigning intensity values
    count = 0
    intensity = 0
    
    # iterating through the sorted array, and assigning intensity values to
    # correct pixels (First 256 pixels have intensity=0, next 256 have
    # intensity=1, etc.)
    for i in np.arange(arr.size):
        x,y = x_y(arr[i])
        img[y][x] = intensity
        count += 1
        if (count >= 256):
            count = 0
            intensity += 1
    
    return img

# computes the symmetric diffeomorphic transformations between the test image and the training image, using the
# dipy library. returns the mapping that minimizes transformation between train and test
def nl_transform(train, test):
    static = test  # test is the static/fixed image, which we want the training image to align with
    moving = train # train is the moving image that we will be applying the transformation to
    
    # constructing the optimization metric we want to minimize, which in this case is cross-correlation
    sigma_diff = 3.0
    radius     = 4
    metric     = CCMetric(2, sigma_diff, radius)
#     metric = SSDMetric(2)
    
    # number of levels, and iterations at each level of the Gaussian pyramid
    level_iters = [100, 50, 25]
    sdr = SymmetricDiffeomorphicRegistration(metric, level_iters)
#     sdr.callback = callback_CC # displays overlayed images at each stage in optimization
    
    # optimizing the transformation between the static and moving image using the above parameters
    mapping = sdr.optimize(static, moving)
    return mapping

# returns jaccard index, which we want to maximize for certain ROIs
def jaccard_index(manual, auto, ROI):
    
    rows = manual.shape[0]
    cols = manual.shape[1]
    
    intersect = 0
    union = 0
    
    for i in np.arange(rows):
        for j in np.arange(cols):
            if (manual[i][j]==ROI and auto[i][j]==ROI):
                intersect += 1
                union += 1
            elif (manual[i][j]==ROI or auto[i][j]==ROI):
                union += 1
    return intersect/union

# using majority label fusion
def majority_label_fusion(arr):
    a = arr[0]
    b = arr[1]
    c = arr[2]
    d = arr[3]
    e = arr[4]
    f = arr[5]
    
    rows, cols = a.shape
    
    output = np.zeros(a.shape)
    
    for i in np.arange(rows):
        for j in np.arange(cols):
            # the output value at each index is the most common label from
            # each of the 6 transformed segmentation images
            list = [a[i][j], b[i][j], c[i][j], d[i][j], e[i][j], f[i][j]]
            output[i][j] = stats.mode(list)[0][0]
    return output

# computes the sum of squared errors between two images
def img_diff(train, val, method):
    
    rows, cols = train.shape
    
    sum = 0
    for i in np.arange(rows):
        for j in np.arange(cols):
            sum += diff(train[i][j], val[i][j], method)
            
    return sum

def diff(x, y, method):
    if (method=="squared"):
        return (x-y)**2
    elif (method=="abs"):
        return np.abs(x-y)

# converts binary segmentation image into RLE format (taken from ECE 4250 Kaggle competition page)
def rle_encode(input, value):
    '''
    input: numpy array, 1 - mask, 0 - background
    Returns run length as string formated
    '''
    rows, cols = input.shape
    img = np.zeros(input.shape)
    
    for i in np.arange(rows):
        for j in np.arange(cols):
            if (input[i][j] == value):
                img[i][j] = 1
    
    pixels = img.flatten()
    pixels = np.concatenate([[0], pixels, [0]])
    runs = np.where(pixels[1:] != pixels[:-1])[0] + 1
    runs[1::2] -= runs[::2]
    return ' '.join(str(x) for x in runs)
